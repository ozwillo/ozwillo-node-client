function copyitem(model,from,to,todatacore){
	model.schema.eachPath(
		function(key)
		{
			if(key == '_id' || key=='__v' || key[0]=='_')
				return;
			if(model.schema.paths[key].instance=='Mixed')
			{
				if(from[key])
				{
				if(todatacore)
				to[key]=JSON.stringify(from[key]);
				else
				to[key]=JSON.parse(from[key]); 
				}
			}
			else
			{
				if(from[key])
				to[key]=from[key];
			}

		});
}
